# Setup logging
def setup_logging():
    """Setup logging configuration"""
    LOG_DIR = os.path.join(os.path.dirname(os.path.dirname(__file__)), 'logs')
    os.makedirs(LOG_DIR, exist_ok=True)
    
    log_filename = f"sharepoint_upload_{datetime.now().strftime('%Y%m%d_%H%M%S')}.log"
    log_path = os.path.join(LOG_DIR, log_filename)
    
    logging.basicConfig(
        level=logging.INFO,
        format='%(asctime)s - %(levelname)s - %(message)s',
        handlers=[
            logging.FileHandler(log_path),
            logging.StreamHandler(sys.stdout)
        ]
    )
    
    logger = logging.getLogger(__name__)
    logger.info(f"SharePoint uploader logging initialized. Log file: {log_path}")
    return logger

logger = setup_logging()

class SharePointUploader:
    """
    SharePoint uploader with tenant-compatible authentication
    """
    
    def __init__(self):
        """Initialize with configuration"""
        self.tenant_id = os.getenv('SHAREPOINT_TENANT_ID')
        self.client_id = os.getenv('SHAREPOINT_CLIENT_ID') 
        self.client_secret = os.getenv('SHAREPOINT_CLIENT_SECRET')
        
        # Optional: Service account credentials for delegated flow
        self.service_username = os.getenv('SHAREPOINT_SERVICE_USERNAME')  # Optional
        self.service_password = os.getenv('SHAREPOINT_SERVICE_PASSWORD')  # Optional
        
        self.site_url = os.getenv('SHAREPOINT_SITE_URL', 'https://jeanalytics.sharepoint.com/sites/LSTM')
        self.folder_path = os.getenv('SHAREPOINT_FOLDER_PATH', 'Shared Documents')
        
        # Extract components
        if 'sharepoint.com' in self.site_url:
            url_parts = self.site_url.replace('https://', '').split('/')
            self.tenant_name = url_parts[0].split('.')[0]
            self.hostname = url_parts[0]
            self.site_name = url_parts[-1] if len(url_parts) > 2 else 'root'
        else:
            raise ValueError(f"Invalid SharePoint site URL: {self.site_url}")
        
        self.token_url = f"https://login.microsoftonline.com/{self.tenant_id}/oauth2/v2.0/token"
        self.graph_base_url = "https://graph.microsoft.com/v1.0"
        self.access_token = None
        
        logger.info(f"🚀 SharePoint uploader initialized")
        logger.info(f"   Site: {self.site_url}")
        logger.info(f"   Tenant blocks app-only tokens - using workaround")
        
    def validate_config(self) -> bool:
        """Validate configuration"""
        required_vars = {
            'SHAREPOINT_TENANT_ID': self.tenant_id,
            'SHAREPOINT_CLIENT_ID': self.client_id, 
            'SHAREPOINT_CLIENT_SECRET': self.client_secret
        }
        
        missing_vars = [var for var, value in required_vars.items() if not value]
        
        if missing_vars:
            logger.error(f"❌ Missing environment variables: {missing_vars}")
            return False
            
        logger.info("✅ Configuration validated")
        return True
    
    def try_device_code_flow(self) -> bool:
        """
        Attempt device code flow (interactive, but can work in some automation scenarios)
        """
        logger.info("🔐 Attempting device code flow")
        
        device_code_url = f"https://login.microsoftonline.com/{self.tenant_id}/oauth2/v2.0/devicecode"
        
        device_data = {
            'client_id': self.client_id,
            'scope': 'https://graph.microsoft.com/Sites.ReadWrite.All offline_access'
        }
        
        try:
            response = requests.post(device_code_url, data=device_data)
            if response.status_code == 200:
                device_info = response.json()
                
                user_code = device_info.get('user_code')
                device_code = device_info.get('device_code')
                verification_uri = device_info.get('verification_uri')
                expires_in = device_info.get('expires_in', 900)
                
                logger.info(f"📱 Device code authentication required:")
                logger.info(f"   Go to: {verification_uri}")
                logger.info(f"   Enter code: {user_code}")
                logger.info(f"   This code expires in {expires_in} seconds")
                
                # For automation, this won't work, but log the info
                logger.warning("⚠️  Device code flow requires manual intervention - skipping")
                return False
                
            else:
                logger.warning("⚠️  Device code request failed")
                return False
                
        except Exception as e:
            logger.warning(f"⚠️  Device code flow failed: {e}")
            return False
    
    def try_certificate_auth(self) -> bool:
        """
        Try certificate-based authentication (if tenant allows)
        """
        logger.info("🔐 Attempting certificate-based auth")
        
        # This would require certificate setup - placeholder for now
        logger.info("⚠️  Certificate auth requires additional setup - skipping")
        return False
    
    def try_legacy_sharepoint_auth(self) -> bool:
        """
        Try legacy SharePoint authentication methods
        """
        logger.info("🔐 Attempting legacy SharePoint authentication")
        
        # Try SharePoint App-Only with legacy endpoint
        legacy_token_url = f"https://accounts.accesscontrol.windows.net/{self.tenant_id}/tokens/OAuth/2"
        
        # SharePoint principal
        resource = f"00000003-0000-0ff1-ce00-000000000000/{self.hostname}@{self.tenant_id}"
        client_id_full = f"{self.client_id}@{self.tenant_id}"
        
        token_data = {
            'grant_type': 'client_credentials',
            'client_id': client_id_full,
            'client_secret': self.client_secret,
            'resource': resource
        }
        
        try:
            response = requests.post(legacy_token_url, data=token_data)
            
            if response.status_code == 200:
                token_response = response.json()
                access_token = token_response.get('access_token')
                
                if access_token:
                    logger.info("✅ Legacy SharePoint auth successful")
                    self.access_token = access_token
                    return True
                    
            logger.warning(f"⚠️  Legacy auth failed: HTTP {response.status_code}")
            if response.text:
                logger.warning(f"Response: {response.text}")
            return False
            
        except Exception as e:
            logger.warning(f"⚠️  Legacy auth exception: {e}")
            return False
    
    def test_token_with_graph(self) -> bool:
        """
        Test the token with Microsoft Graph API
        """
        logger.info("🧪 Testing token with Microsoft Graph API")
        
        headers = {
            'Authorization': f'Bearer {self.access_token}',
            'Content-Type': 'application/json'
        }
        
        # Test with Graph API sites endpoint
        test_url = f"{self.graph_base_url}/sites/{self.hostname}:/sites/{self.site_name}"
        
        try:
            response = requests.get(test_url, headers=headers)
            
            if response.status_code == 200:
                site_data = response.json()
                site_id = site_data.get('id')
                display_name = site_data.get('displayName', 'Unknown')
                
                logger.info(f"✅ Graph API access successful")
                logger.info(f"   Site: {display_name} (ID: {site_id})")
                return True
                
            else:
                logger.error(f"❌ Graph API test failed: HTTP {response.status_code}")
                logger.error(f"Response: {response.text}")
                return False
                
        except Exception as e:
            logger.error(f"❌ Graph API test exception: {e}")
            return False
    
    def test_token_with_rest(self) -> bool:
        """
        Test the token with SharePoint REST API
        """
        logger.info("🧪 Testing token with SharePoint REST API")
        
        headers = {
            'Authorization': f'Bearer {self.access_token}',
            'Accept': 'application/json;odata=verbose',
            'Content-Type': 'application/json'
        }
        
        test_url = f"{self.site_url}/_api/web"
        
        try:
            response = requests.get(test_url, headers=headers)
            
            if response.status_code == 200:
                site_data = response.json()
                site_title = site_data.get('d', {}).get('Title', 'Unknown')
                
                logger.info(f"✅ SharePoint REST API access successful")
                logger.info(f"   Site: {site_title}")
                return True
                
            else:
                logger.error(f"❌ REST API test failed: HTTP {response.status_code}")
                logger.error(f"Response: {response.text}")
                return False
                
        except Exception as e:
            logger.error(f"❌ REST API test exception: {e}")
            return False
    
    def get_working_token(self) -> bool:
        """
        Try multiple authentication methods to get a working token
        """
        logger.info("🔑 Attempting multiple authentication methods")
        
        auth_methods = [
            ("Legacy SharePoint Auth", self.try_legacy_sharepoint_auth),
            ("Certificate Auth", self.try_certificate_auth),
            ("Device Code Flow", self.try_device_code_flow),
        ]
        
        for method_name, method_func in auth_methods:
            logger.info(f"🔄 Trying: {method_name}")
            
            if method_func():
                logger.info(f"✅ {method_name} successful")
                
                # Test the token
                if self.test_token_with_graph() or self.test_token_with_rest():
                    logger.info(f"✅ Token validated with {method_name}")
                    return True
                else:
                    logger.warning(f"⚠️  {method_name} got token but validation failed")
            else:
                logger.info(f"❌ {method_name} failed")
        
        logger.error("❌ All authentication methods failed")
        return False
    
    def upload_csv_files(self) -> bool:
        """
        Main upload process
        """
        logger.info("🚀 Starting CSV upload with tenant workarounds")
        
        # Validate config
        if not self.validate_config():
            return False
        
        # Get working token
        if not self.get_working_token():

            logger.error("")

            logger.error("   Your SharePoint tenant blocks 'app-only' authentication.")
            logger.error("   This is a common security setting in many organizations.")

        # If we get here, we have a working token
